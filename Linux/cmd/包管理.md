### yum
    #默认源
    /etc/yum.repos.d/
    #谁提供了这个命令  
    yum provides locate
    #列出现有的yum源
    yum repolist
    #清除原有yum数据信息
    yum clean all
    #更新yum库信息
    yum makecache

### 换源

    mv /etc/yum.repos.d/CentOS-Base.repo /etc/yum.repos.d/CentOS-Base.repo.backup
    wget  -P http://mirrors.163.com/.help/CentOS7-Base-163.repo /etc/yum/repos.d/CentOS-Base.repo
    yum clean all
    yum makecache

### 查询

    yum list | grep chinese
    #查找名字和描述
    yum search  {keyword}
    #谁提供了这个命令  
    yum provides locate
    #查找vsftpd属于哪个包
    yum search */vsftpd.conf

### 卸载

    yum -y remve httpd
    yum history 
    yum history info
    #将依赖也卸载
    yum history undo 6
